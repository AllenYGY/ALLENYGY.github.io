无重复字符的最长子串

class Solution {
public:
	int lengthOfLongestSubstring(string s) {
		int ans=0,left=0,n=s.size();
		unordered_map<char,int>m;
		for(int right=0;right<n;right++){
			m[s[right]]++;
			while(m[s[right]]>1){
				m[s[left]]--;
				left++;
			}ans=max(ans,right-left+1);
		}	
		return ans;
	}
};
